  "saghen/blink.cmp",

      "MahanRahmati/blink-nerdfont.nvim",

    sources = {
      default = {
        ...
        "nerdfont",
      },
      providers = {
        nerdfont = {
          module = "blink-nerdfont",
          name = "Nerd Fonts",
          score_offset = 15, -- Tune by preference
          opts = { insert = true }, -- Insert nerdfont icon (default) or complete its name
        }
      }
    }
  }
}
